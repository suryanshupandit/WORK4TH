9. Write a program to: - Take input string - Count vowels and consonants 
ans-:
text = input("Enter a string: ").lower()
vowels = "aeiou"
v_count = 0
c_count = 0

for char in text:
    if char.isalpha():
        if char in vowels:
            v_count += 1
        else:
            c_count += 1

print(f"Vowels: {v_count}, Consonants: {c_count}")

with open("sample.txt", "r") as f:
    lines = f.readlines()
    line_count = len(lines)
    word_count = sum(len(line.split()) for line in lines)
    char_count = sum(len(line) for line in lines)

print(f"Lines: {line_count}, Words: {word_count}, Characters: {char_count}")

11. Write a program: - Create a class BankAccount - Methods: deposit, withdraw, check balance 
ans-:
class BankAccount:
    def __init__(self, balance=0):
        self.balance = balance
    def deposit(self, amount):
        self.balance += amount
        print(f"Deposited {amount}. New balance: {self.balance}")
    def withdraw(self, amount):
        if amount <= self.balance:
            self.balance -= amount
            print(f"Withdrew {amount}. New balance: {self.balance}")
        else:
            print("Insufficient funds!")
    def check_balance(self):
        print(f"Current Balance: {self.balance}")
12. Write a program: - Accept list of numbers - Remove duplicates 
- Sort it without using sort() 
ans-:
nums = [4, 2, 5, 2, 1, 4, 3]
# Remove duplicates
unique_nums = list(set(nums))
# Bubble Sort
n = len(unique_nums)
for i in range(n):
    for j in range(0, n - i - 1):
        if unique_nums[j] > unique_nums[j + 1]:
            unique_nums[j], unique_nums[j + 1] = unique_nums[j + 1], unique_nums[j]

print(unique_nums)

13. Write a program using lambda + map + filter: - Square only even numbers from list 
SECTION D: Advanced / Thinking 
ans-:
nums = [1, 2, 3, 4, 5, 6]
# Filter even, then square them
result = list(map(lambda x: x**2, filter(lambda x: x % 2 == 0, nums)))
print(result) # Output: [4, 16, 36]

19. Python + SQL: - Connect database - Create table Student 
- Insert 3 records - Fetch and display all 
ans-:
import sqlite3
conn = sqlite3.connect("school.db")
cursor = conn.cursor()
cursor.execute("CREATE TABLE IF NOT EXISTS Student (id INT, name TEXT)")
cursor.executemany("INSERT INTO Student VALUES (?, ?)", [(1, 'Alice'), (2, 'Bob'), (3, 'Charlie')])
cursor.execute("SELECT * FROM Student")
for row in cursor.fetchall():
    print(row)
conn.close()
